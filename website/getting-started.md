## Requirements

- Node.js 20+
- npm 10+

## Install

```bash
npm install pptx-parser-for-ai
```

## Minimal file-based generation

```ts
import { PPTXRenderer } from "pptx-parser-for-ai";

const renderer = new PPTXRenderer({ enableQA: true });
await renderer.generateFromFile("./presentation.yaml", "./output.pptx");
```

## Minimal object-based generation

```ts
import { PPTXRenderer, type PresentationDSL } from "pptx-parser-for-ai";

const dsl: PresentationDSL = {
  version: "1.0",
  theme: "corporate-blue",
  metadata: { title: "Sample" },
  slides: [
    {
      type: "content",
      title: "Hello",
      content: [{ type: "text", content: "Generated by API" }]
    }
  ]
};

await new PPTXRenderer().generate(dsl, "./output.pptx");
```
